    end

      P1[FSPL_los from f_mhz and d3d_los_km]
      P2[Two ray coherent LoS term]
      P3[NLoS term via COST231 A2G branch]
      P4[raw_prior_db from LoS and NLoS hybrid]
      P1 --> P2
      P2 --> P4
      P3 --> P4
    end

    subgraph CAL[Regime Calibration]
      C1[Detect regime from LoS and obstruction]
      C2[Per regime polynomial y equals a2 x squared plus a1 x plus a0]
      C3[y_los and y_nlos from raw_prior_db]
      C4[prior_cal_db equals p_los y_los plus one minus p_los y_nlos]
      C5[Clip prior_cal_db to min and max]
      C1 --> C2 --> C3 --> C4 --> C5
    end

    subgraph USE[Residual Learning Usage]
      U1[x_prior equals normalize prior_cal_db]
      U2[r_star equals y_true_db minus prior_cal_db]
      U3[Network predicts residual r_hat]
      U4[y_hat_db equals prior_cal_db plus r_hat]
      U1 --> U2 --> U3 --> U4
    end
